IMPORT

IMPORT

Import всього модуля

IMPORT

Aliasing та Side Effects

// Перейменування імпорту

EXPORT

EXPORT

Іменований експорт

// Експорт під час оголошення
export const PI = 3.14159;
export function add(a, b) { return a + b; }
// Експорт існуючих змінних
export { myFunction, anotherVar };

EXPORT

Дефолтний експорт

// Функція
export default function Calculator() {}
// Клас
export default class MyClass {}
// Значення
export default 42;

EXPORT

Re-export (Переекспорт)

// Переекспорт всього
export * from './math';
// Переекспорт конкретних функцій
export { add, subtract } from './math';
// Переекспорт з перейменуванням
export { default as Calculator } from './Calculator';

COMMONJS vs ESM

CommonJS (Node.js)

// Export
module.exports = { add };
exports.PI = 3.14;

// Import
const { add } = require('./math');
const math = require('./math');

ESM (ES6 Modules)

// Export
export { add };
export const PI = 3.14;

// Import
import { add } from './math';
import * as math from './math';

TREE SHAKING

Що це таке?

  • Видалення невикористаного коду з бандлу
  • Зменшення розміру фінального файлу
  • Працює тільки з ES6 модулями
  • Статичний аналіз import/export

TREE SHAKING

utils.js

export function add(a, b) {
  return a + b;
}

export function subtract(a, b) {
  return a - b;
}

export function multiply(a, b) {
  return a * b;
}

main.js

// Імпортуємо тільки add
